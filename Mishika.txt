name: Merge DDIQAGH to DDIUATGH

on:
  workflow_dispatch:

permissions:
  contents: write
  pull-requests: write
  checks: read
  actions: write

jobs:
  promote:
    runs-on: uhg-runner-m
    environment: DDIUATGH
    steps:
      - name: Checkout target branch (DDIUATGH)
        uses: actions/checkout@v4
        with:
          ref: DDIUATGH
          token: ${{ secrets.PAT_TOKEN }}
          fetch-depth: 0

      - name: Configure Git
        run: |
          git config user.email "maltss_tech@optum.com"
          git config user.name "maltssdeploy"

      - name: Add remote and fetch source branch (DDIQAGH)
        run: |
          git remote add source-origin https://x-access-token:${{ secrets.PAT_TOKEN }}@github.com/${{ github.repository }}.git || true
          git fetch source-origin DDIQAGH

      # Get last promoted SHA
      - name: Get last promoted SHA
        id: lastsha
        run: echo "sha=${{ vars.LAST_GOOD_SHA_DDIUATGH_MERGE }}" >> $GITHUB_OUTPUT

      - name: Find changed files since last promotion
        run: |
          mkdir -p artifacts
          if [ -n "${{ steps.lastsha.outputs.sha }}" ]; then
            git diff --name-status ${{ steps.lastsha.outputs.sha }} source-origin/DDIQAGH -- force-app/main/default/ > artifacts/changed_files.txt
          else
            git diff --name-status origin/DDIUATGH source-origin/DDIQAGH -- force-app/main/default/ > artifacts/changed_files.txt
          fi
          echo "Changed files:"
          cat artifacts/changed_files.txt

      - name: Generate package.xml and destructiveChanges.xml
        run: |
          mkdir -p artifacts
          # Very basic XML generation for demonstration
          echo '<?xml version="1.0" encoding="UTF-8"?>' > artifacts/package.xml
          echo '<Package xmlns="http://soap.sforce.com/2006/04/metadata">' >> artifacts/package.xml
          echo '  <types>' >> artifacts/package.xml
          echo '    <!-- Add components here -->' >> artifacts/package.xml
          echo '  </types>' >> artifacts/package.xml
          echo '  <version>59.0</version>' >> artifacts/package.xml
          echo '</Package>' >> artifacts/package.xml

          echo '<?xml version="1.0" encoding="UTF-8"?>' > artifacts/destructiveChanges.xml
          echo '<Package xmlns="http://soap.sforce.com/2006/04/metadata">' >> artifacts/destructiveChanges.xml
          echo '  <!-- Add deletions here -->' >> artifacts/destructiveChanges.xml
          echo '</Package>' >> artifacts/destructiveChanges.xml

      - uses: actions/upload-artifact@v4
        with:
          name: delta-artifacts
          path: artifacts/*

      # ⬇️ NEW: Replace full merge with path-specific promotion
      - name: Checkout only force-app/main/default from DDIQAGH
        run: |
          git checkout source-origin/DDIQAGH -- force-app/main/default

      - name: Commit changes (if any)
        run: |
          if [ -n "$(git status --porcelain)" ]; then
            git add force-app/main/default
            git commit -m "Promote force-app/main/default from DDIQAGH to DDIUATGH"
          else
            echo "No changes to commit"
          fi

      - name: Push changes to DDIUATGH
        run: git push origin HEAD:DDIUATGH

      - name: Update SHA ID 
        run: |
          NEW_SHA=$(git rev-parse source-origin/DDIQAGH)
          gh api \
            --method PATCH \
            -H "Accept: application/vnd.github+json" \
            /repos/${{ github.repository }}/environments/DDIUATGH/variables/LAST_GOOD_SHA_DDIUATGH_MERGE \
            -f name="LAST_GOOD_SHA_DDIUATGH_MERGE" \
            -f value="$NEW_SHA"
        env:
          GITHUB_TOKEN: ${{ secrets.PAT_TOKEN }}

      - name: Success message
        run: echo "✅ Successfully promoted force-app/main/default from DDIQAGH to DDIUATGH"
